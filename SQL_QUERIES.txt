SELECT 
    s.name AS StudentName,
    t.teacher_name AS TeacherName
FROM 
    Student s
JOIN 
    StudentCourse sc ON s.student_id = sc.student_id
JOIN 
    CourseEdition ce ON sc.course_edition_id = ce.course_edition_id
JOIN 
    Course c ON ce.course_id = c.course_id
JOIN 
    TeacherCourseEdition tce ON ce.course_edition_id = tce.course_edition_id
JOIN 
    Teacher t ON tce.teacher_id = t.teacher_id
WHERE 
    c.workload < 60;

WITH StudentCourseCount AS (
    SELECT student_id
    FROM StudentCourse
    GROUP BY student_id
    HAVING COUNT(*) > 2
)
SELECT 
    s.name,
    TO_CHAR(sc.enrollment_date, 'Day') as day_of_week,
    TO_CHAR(sc.enrollment_date, 'Month') as month,
    sc.enrollment_date as full_date,
    TO_CHAR(sc.enrollment_date, 'HH24:MI') as enrollment_time
FROM Student s
JOIN StudentCourse sc ON s.student_id = sc.student_id
JOIN StudentCourseCount scc ON s.student_id = scc.student_id
WHERE 
    TO_CHAR(sc.enrollment_date, 'HH24:MI') > '12:30'
    AND sc.enrollment_date < ADD_MONTHS(SYSDATE, -36)
ORDER BY 
    sc.enrollment_date DESC,
    TO_CHAR(sc.enrollment_date, 'HH24:MI') ASC;
